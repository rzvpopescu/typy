ViewModel : 
                    import {Router,Route} from 'Router';
                    export default class MyPage{

                        router:Router;

                        constructor(){
                            this.router =  new Router();
                            this.router.addRoute(new Route(['','home'],'app/path/to/home','Home'));
                            this.router.addRoute(new Route(['about'],'app/path/to/about','About'));
                            this.router.addRoute(new Route(['params/:param1/:param2'], 'app/path/to/params', 'Route with parameters'));
                        }
                    }
