For binding - Advanced

Syntax : for.bind='item of items'

Students:
  • ${student.name} - ${student.age} -
ViewModel : 
                        class Student{
                            name:String;
                            age:number;                            
                            constructor(name:string,age:number) {
                                this.name = name;
                                this.age = age;                
                            }
                        }

                        export default class MyPage{
                            students:Array =  new Array();
                            constructor(){
                                this.populateStudents();
                            }
                            populateStudents(){
                                this.students.push(new Student('John',22));
                                this.students.push(new Student('Mary',21));
                                this.students.push(new Student('Agnes',20));
                                this.students.push(new Student('Edward',24));
                            }
                            addStudent(){
                                this.students.push(new Student(prompt("Introduce name"),parseInt(prompt("Introduce age"))));
                            }
                             deleteStudent(student:Student){
                                this.students.splice(this.students.indexOf(student),1);
                            }
                        }
View: 
                          <ul for.bind="student of students">
                                <li>${student.name} - ${student.age} - <button class="btn" click.bind="parent.deleteStudent(student)">Delete</button></li>
                          </ul>
                          <button class="btn" click.bind="addStudent()">Add student</button>

If/IfNot binding

Syntax : if.bind='someCondition'

Syntax : ifnot.bind='someCondition'

[1] I'm rendered when 'canRender' flag is "true".
[2] I'm rendered when 'canRender' flag is "false".
ViewModel: 
                    export default class MyPage{

                    canRender:boolean =  false;
                    changeRender(){
                            this.canRender = !this.canRender;
                        }
                    }
View: 
                <div if.bind="canRender">
                    <span>I'm rendered when 'canRender' is "true".</span>
                </div>
                <div ifnot.bind="canRender">
                    <span>I'm rendered when 'canRender' is "false".</span>
                </div>
                <button click.bind="changeRender()">Toggle 'canRender'</button>

Viewport binding

Syntax : viewport.bind='routerInstance'

ViewModel: 
                    import {Router,Route} from 'Router';
                    export default class MyPage{

                        router:Router;

                        constructor(){
                            this.router =  new Router();
                            this.router.addRoute(new Route(['','home'],'app/path/to/home','Home'));
                            this.router.addRoute(new Route(['about'],'app/path/to/about','About'));
                        }
                    }
View: 
                    <a href="#home">Home</a>
                    <a href="#about">About</a>
                    <div view-port.bind='router'></div>
